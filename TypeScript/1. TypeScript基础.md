#### TypeScript

TypeScript是JavaScript的超集，是编译型的语言。执行前会通过编译器进行预编译

TypeScript的特点：
-   类型注解 和 编译时类型检查
-   基于类的面向对象编程
-   泛型
-   接口
-   声明文件

类型注解：
```
/* 类型约束 - 常用内置类型 string, number, boolean, any, void */
// 声明tile变量是string类型
let title: string;
title = 'xx';


/* 类型推断 */
// 默认推断name是字符串类型
let name = 'xx';


/* 数组的类型约束 */
// 单种的类型约束：name是字符串数组，此处1会报错
let name: string[];
name = ['json', 'tom', 1];

// 多种的类型约束：name可以数值也可以是字符串
let name: (string | number)[];
name = ['json', 'tom', 1];

// 任意类型的约束
let list: any[];
list = [1, true, 'free']

// 在函数中类型约束：约束形参和返回值的类型
function greeting(person: string): string {
    return 'hellow' + person;
}
greeting('tom')

// 无返回值 void类型
function warn(): void {
    alert('warning!');
}


/* ts函数中如果声明了参数类型，就是必选参数 */
// name是必选参数，age是可选参数
function sayHello(name: string, age?: number): string {
    return `${name} ${age}`
}
sayHello('tom', 20)

// 对参数指定默认值, 返回值可以是字符串或数值
function sayHello(name: string, age: number = 20): string | number {
    return `${name} ${age}`
}
sayHello('tom')


/* 函数重载：多个同名函数，通过参数数量或者类型不同返回值不同 */
// ts中先声明再创建
function info(a: object): string;
function info(a: string): object;

function info(a: any): any {
    if (typeof a === 'object') {
        return a.name
    } else {
        return { name: a }
    }
}
info({ name: 'lida' })
info('lida')
```

```
/* 在vue中使用 */
<template>
    <div>
        <ul>
            <input type="text" @keyup.enter="addItem">
            <li v-for="item in list" :key="item">{{item}}</li>
        </ul>
    </div>
</template>
<script lang="ts">
    import {Vue, Component, Prop} from 'vue'
    import Alert from './components/Alert.vue'
    
    @Component({
        components: {
            Alert
        }
    })

    // 用Component增强当前类
    @Component 
    export default class Index extends Vue {
        // props 属性 msg 是必填项，而且是字符串类型：! 必选属性，? 可选属性
        // 变量的作用域：private(私有的)，public(公有的)，protected(受保护的)
        @Prop() private msg!: string; 
        
        // 对属性添加默认值
        @Prop() private name: string = 'lida';

        // 传入对对象必须有键foo值为字符串
        @Prop() private obj: { foo: string };


        // 普通属性相当于data中对属性
        private list: string[] = ['静态类型', '编译']

        
        // 添加方法
        addItem(event: any) {
            this.list.push(event.target.value)

            event.target.value = ''
        }
    }
</script>
```
