#### 中序遍历
![遍历结构](https://pic.imgdb.cn/item/5f1fdb5e14195aa594b3bbbe.png)

```
const tree = {
    value: 1,
    left: {
        value: 2,
        left: {
            value: 4
        },
        right: {
            value: 5
        }
    },
    right: {
        value: 3,
        left: {
            value: 6
        },
        right: {
            value: 7
        }
    }
}
// 中序遍历
const inorder = root => {
    /**
     * 下图的执行顺序：
     * 因为函数执行是后进先出原则，最后进栈最先出栈。
    */
    if (!root) return
    console.log(root)           // 1,2,4,5,3,6,7

    inorder(root.left)
    console.log(root.value)     // 4,2,5,1,6,3,7
    inorder(root.right)
}
inorder(tree)
```
