#### 后序遍历
![遍历结构](https://pic.imgdb.cn/item/5f1fdb7d14195aa594b40b8a.png)

```
const tree = {
    value: 1,
    left: {
        value: 2,
        left: {
            value: 4
        },
        right: {
            value: 5
        }
    },
    right: {
        value: 3,
        left: {
            value: 6
        },
        right: {
            value: 7
        }
    }
}

const postorder = (root) => {
    if (!root) return
    postorder(root.left)
    postorder(root.right)
    console.log(root.value)
}
postorder(tree)
```
